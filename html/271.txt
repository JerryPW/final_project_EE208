    function heartBeat(){
        var diffY=0;
        if (document.documentElement && document.documentElement.scrollTop){
            diffY = document.documentElement.scrollTop;
        }else if (document.body){
            diffY = document.body.scrollTop
        }

        var percent=diffY-lastscroll;

        document.getElementById("leftDIV1").style.top=parseInt(document.getElementById("leftDIV1").style.top)+percent+"px";
        document.getElementById("rightDIV1").style.top=parseInt(document.getElementById("rightDIV1").style.top)+percent+"px";

        document.getElementById("rightDIV2").style.top=parseInt(document.getElementById("rightDIV2").style.top)+percent+"px";

        function distinctTimeArr(timeArr) {
           var result = new Array();
           for (var i = 0; i < timeArr.length; i++) {
               var item = timeArr[i];
               if (result.indexOf(item) === -1) {
            	   result.push(item);
               }
           }
           return result;
       }
