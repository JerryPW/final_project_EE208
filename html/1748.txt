        function getTimeArr(json) {
            var timeArr = new Array();
            for(var i = 0; i < json.length; i++) {
            	if(json[i]) {
                    var s = json[i].jzsjRq;
                    //s = new Date(s);

                    var ss = s.split(" ").length > s.split("$nbsp;").length ? s.split(" ") : s.split("$nbsp;");
                    timeArr[i] = ss[0];
                }
            }
            return timeArr;
