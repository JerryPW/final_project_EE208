// 切换语言功能
function switch_lang()
{
	$(".link-lang").click(function(){
		var lang = $(this).attr('lang');
		if ($.cookie('lang') == lang)
		{
			return false;
		}
		$.cookie('lang', lang);
		location.reload();//刷新当前窗口
		return false;
	});
}

砥砺前行创佳绩 乘势而上再出发——校园管理办公室党支部开展九月学习交流会
时间：2022-09-29 07:19
为了深入学习中共上海交通大学第十一届委员会第八次全体会议报告，凝心聚力推动校园管理事业发展，2022年9月29日上午，校园管理办公室党支部举行了九月线上学习交流会。会议由党支部书记王震主持，全体党员参加。党支部书记王震带领支部党员学习了中共上海交大第十一届委员会第八次全体会议报告。报告以《砥砺前行深化改革 奋发有为再创佳绩》为题，从加强党的全面领导，为打赢疫情防控阻击战提供坚强政治保证；坚持统筹兼顾，实现各项事业快速平稳发展两大方面，回顾总结上半年的主要工作。同时，进一步明确秋季学期的工作思路及主要任务：要坚定信心攻坚克难，聚焦重点深化改革，弘扬交大精神品格；坚持以人为本，提升师生关心服务水平；聚焦瓶颈问题，强化校园精细化管理。王震强调，要统筹做好常态化疫情防控工作，联动办公室线上平台和线下服务，织牢防控网络。支部党员结合疫情防控相关要求，在会上交流了近期工作内容和未来工作安排。通过此次学习交流会，支部党员更加明白学校今年取得的卓越成果来之不易，未来将进一步坚持育人初心，把“我为群众办实事”作为常态化工作开展；持续推动改革创新，不断提升校园管理水平，以实际行动迎接党的二十大胜利召开。
