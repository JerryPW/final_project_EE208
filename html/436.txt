// 切换语言功能
function switch_lang()
{
	$(".link-lang").click(function(){
		var lang = $(this).attr('lang');
		if ($.cookie('lang') == lang)
		{
			return false;
		}
		$.cookie('lang', lang);
		location.reload();//刷新当前窗口
		return false;
	});
}

从严治党守初心，凝心铸魂落实效——校园管理办公室党支部召开6月专题学习交流会
时间：2022-07-05 06:50
为了深入学习领会习近平总书记在十九届中共中央政治局第四十次集体学习时的重要讲话精神，加深党风廉政建设和反腐败斗争的认识，2022年6月30日，校园管理办公室党支部举行了6月专题学习交流会。会议由党支部书记王震主持，全体党员参加。党支部书记王震传达了中共中央政治局第四十次集体学习会议的会议精神。习近平总书记的重要讲话，进一步阐明了伟大自我革命和反腐败斗争的重大意义和深刻内涵。会议强调，要加强党中央对反腐败工作的集中统一领导，提高一体推进不敢腐、不能腐、不想腐的能力和水平，全面打赢反腐败斗争攻坚战、持久战。王震指出，本次讲话充分反映了习近平总书记和党中央从严治党的决心，各位党员需在思想认识上高度重视，在关键时刻勇挑重担，用行动服务好师生群众。党支部副书记刘晓婷通报了学校党委和上级党组织有关工作部署和近期工作提示。她指出，党支部要持续巩固党史学习教育和“四史”学习教育成果，发挥战斗堡垒和先锋模范作用。在常态化疫情防控的实际工作中不断强化对疫情防控方针政策的全面认识，积极开展联建共建，为群众解决实际问题、推动实际工作，为疫情防控和校园稳定发展贡献力量。支部全体党员集体观看了微视频《我们是党员突击队》，并就疫情防控期间的工作和暑期工作规划作交流分享。通过建立党支部“第一议题”学习制度，开展支部书记领学，党员交流研讨等环节，让党员们及时跟进学习习近平总书记最新重要讲话精神，进一步加深学习认识。本次议题让党员们加深了对“勇于自我革命是党百年奋斗培育的鲜明品格”的认识，进一步筑牢思想道德防线，强化自我监督。党员们表示更要以身作则，保持政治定力，坚持高标准严要求，持续做好校园管理服务工作。会后，党支部向每位党员发放了《习近平在上海》、《当好改革开放的排头兵——习近平上海足迹》读本，鼓励党员干部持续深入学习贯彻习近平新时代中国特色社会主义思想，坚持不懈用习近平新时代中国特色社会主义思想凝心铸魂。
