// 切换语言功能
function switch_lang()
{
	$(".link-lang").click(function(){
		var lang = $(this).attr('lang');
		if ($.cookie('lang') == lang)
		{
			return false;
		}
		$.cookie('lang', lang);
		location.reload();//刷新当前窗口
		return false;
	});
}

勇担使命共抗疫，坚决打赢攻坚战——校园管理办公室党支部召开5月专题学习交流会
时间：2022-05-16 10:40
为了深入学习领会5月5日习近平总书记主持召开的中共中央政治局常委会会议精神，进一步坚定信心打好大上海保卫战，2022年5月8日，校园管理办公室党支部聚焦现阶段疫情防控重点工作，举行了5月专题学习交流会。会议由党支部书记王震主持，全体党员参加。王震首先带领全体党员学习了习近平总书记主持召开的中共中央政治局常务委员会会议精神及上海市委常委会扩大会议精神，会议指出，经过全国上下勠力同心、并肩作战，我们经受住了武汉保卫战以来最为严峻的防控考验，取得了阶段性成效。会议强调，疫情防控工作正处于关键时期和吃劲阶段，要坚定信心，坚决落实党中央决策部署，全面动员、全面部署，坚决巩固来之不易的疫情防控成果，做到守土有责、守土尽责。各位党员先后结合疫情防控交流了近期的工作重点，并分享了在校内或社区的志愿服务工作感想。本轮疫情发生以来，校园管理办公室在校党委的领导下，充分发挥基层党支部的战斗堡垒作用，多位同志先后逆行入校参与校园防疫工作，隔离在家的同志积极投身社区志愿者工作。通过此次学习，支部党员深刻认识到抗疫斗争的复杂性和艰巨性，进一步增强了责任感和紧迫感。全体党员表示，要持续以时不我待的精神，抓实抓细校园管理中的疫情防控各项工作，持续坚持“动态清零”总方针不动摇，坚决筑牢疫情防控屏障。
